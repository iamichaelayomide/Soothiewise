structured 5-7 day care plan with specific daily activities.

  "supportNeed": string,
  "primaryConcerns": string[],

  "tonePreference": string,
  "phq9Score": number,
  "gad7Score": number
}

OUTPUT (strict JSON):
{
  "planTitle": string,
  "planSummary": string (2-3 sentences, warm and human),
  "days": [
    {
      "day": number,
      "theme": string,
      "activities": [

          "time": "morning" | "afternoon" | "evening",
          "type": "journal" | "breathwork" | "checkin" | "reflection" | 
                  "reading" | "exercise" | "chat",
          "title": string,
          "description": string,
          "durationMins": number,
          "moduleId": string | null
        }
      ]
    }
  ],
  "therapistRecommendation": boolean,
  "weeklyGoal": string
}

- Include at least one journaling activity every 2 days
